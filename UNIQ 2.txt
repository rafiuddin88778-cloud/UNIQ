let contactsCache = {}; // uid -> profile

function listenContacts() {
  off('contacts');
  const ref = db.ref(`contacts/${currentUser.uid}`);
  const fn = snap => {
    const contacts = snap.val() || {};
    contactsCache = {};
    const list = document.getElementById('contacts-list');
    list.innerHTML = '';
    const uids = Object.keys(contacts).filter(uid => !isBlocked(uid));
    document.getElementById('chats-empty').style.display = uids.length===0 ? 'flex':'none';
    if (uids.length===0) return;

    // Use a map to avoid duplicates
    const rendered = new Set();
    const promises = uids.map(uid => db.ref(`users/${uid}`).once('value'));
    Promise.all(promises).then(snaps => {
      snaps.forEach((s, i) => {
        if (!s.exists()) return;
        const uid = uids[i];
        if (rendered.has(uid)) return;
        rendered.add(uid);
        const p = s.val();
        contactsCache[uid] = p;
        const chatId = getChatId(currentUser.uid, uid);
        db.ref(`messages/${chatId}`).limitToLast(1).once('value', msgSnap => {
          const msgs = msgSnap.val();
          let lastMsg='', lastTime='';
          if (msgs) { const m=Object.values(msgs)[0]; lastMsg=m.text||'📎 Media'; lastTime=formatTime(m.ts); }
          const unread = unreadCounts[chatId]||0;
          const nick = chatNicknames[uid];
          list.appendChild(createContactItem(p, uid, lastMsg, lastTime, unread, nick));
        });
      });
    });
  };
  ref.on('value', fn);
  activeListeners['contacts'] = { ref, event:'value', fn };
}

function createContactItem(profile, uid, lastMsg, lastTime, unread, nick) {
  const div = document.createElement('div');
  div.className = 'list-item';
  div.dataset.uid = uid;
  const displayName = nick || profile.displayName;
  div.innerHTML = `
    <div class="avatar">${getInitials(displayName)}</div>
    <div class="item-body">
      <div class="item-name">${escHtml(displayName)}</div>
      <div class="item-sub">${lastMsg ? escHtml(lastMsg.substring(0,40)) : '@'+profile.username}</div>
    </div>
    <div class="item-meta">
      <span class="item-time">${lastTime}</span>
      ${unread>0 ? `<span class="unread-badge">${unread}</span>` : ''}
    </div>`;
  div.onclick = () => openChat(profile, uid);
  return div;
}

function createInlineSearchItem(u) {
  const div = document.createElement('div');

    <button class="btn-sm" data-uid="${u.uid}">Add</button>`;
  // Check existing state
  db.ref(`contacts/${currentUser.uid}/${u.uid}`).once('value', cs => {
    db.ref(`requests/${u.uid}/${currentUser.uid}`).once('value', rs => {
      const btn = div.querySelector('.btn-sm');

      else if (rs.exists()) { btn.textContent='Requested'; btn.classList.add('sent-state'); }
    });

    e.stopPropagation();

      // Open chat if friends
      openChat(u, u.uid);

async function blockUser(uid) {
  // Send system message before blocking
  if (currentChatId) {
    await sendSystemMessage(`${currentProfile.displayName} blocked you 😔`);
  }
  await db.ref(`users/${currentUser.uid}/blocked/${uid}`).set(true);
  if (currentProfile) { if (!currentProfile.blocked) currentProfile.blocked={}; currentProfile.blocked[uid]=true; }
  await db.ref(`contacts/${currentUser.uid}/${uid}`).remove();
  // Do NOT remove partner's contact entry — they still see the chat
  toast('User blocked.');
  updateProfileModal();
}

async function unblockUser(uid) {
  const partnerSnap = await db.ref(`users/${uid}`).once('value');
  await db.ref(`users/${currentUser.uid}/blocked/${uid}`).remove();
  if (currentProfile?.blocked) delete currentProfile.blocked[uid];
  // Re-add to contacts on both sides
  await db.ref(`contacts/${currentUser.uid}/${uid}`).set(true);
  await db.ref(`contacts/${uid}/${currentUser.uid}`).set(true);
  // Send system message in the chat
  const cid = getChatId(currentUser.uid, uid);
  const tempChatId = currentChatId;
  currentChatId = cid;
  await sendSystemMessage(`${currentProfile.displayName} unblocked you 🥳`);
  currentChatId = tempChatId;
  updateProfileModal();
  toast('User unblocked.');
  listenContacts();
}

async function sendSystemMessage(text) {
  if (!currentChatId) return;
  const msgRef = db.ref(`messages/${currentChatId}`).push();
  await msgRef.set({ id: msgRef.key, text, senderId: '__system__', ts: Date.now(), system: true });
}

// ── Messages listener ────────────────────────
function listenMessages() {
  off('messages');
  const ref = db.ref(`messages/${currentChatId}`);
  const fn = snap => { renderMessages(snap.val()||{}); markUnreadAsRead(); };
  ref.on('value', fn);
  activeListeners['messages'] = { ref, event:'value', fn };
}

function renderMessages(msgs) {
  const container = document.getElementById('chat-messages');
  container.innerHTML = '';
  let lastDate='';
  const sorted = Object.values(msgs).sort((a,b)=>a.ts-b.ts);
  sorted.forEach(m => {
    const d=new Date(m.ts).toLocaleDateString([],{month:'short',day:'numeric',year:'numeric'});
    if (d!==lastDate) {
      lastDate=d;
      const div=document.createElement('div'); div.className='msg-date-divider'; div.textContent=d;
      container.appendChild(div);
    }
    container.appendChild(createMessageEl(m));
  });
  container.scrollTop=container.scrollHeight;
}

function createMessageEl(m) {
  if (m.system) {
    const row=document.createElement('div'); row.className='system-msg'; row.textContent=m.text; return row;
  }
  const isSent = m.senderId===currentUser.uid;
  const row=document.createElement('div');
  row.className=`message-row ${isSent?'sent':'received'}`;
  row.dataset.msgId=m.id;

  let replyHtml='';
  if (m.replyTo) replyHtml=`<div class="reply-preview"><strong>${escHtml(m.replyTo.senderName)}</strong>${escHtml(m.replyTo.text.substring(0,60))}</div>`;

  let reactionsHtml='';
  if (m.reactions) {
    const gr={};
    Object.values(m.reactions).forEach(e=>{gr[e]=(gr[e]||0)+1;});
    reactionsHtml=`<div class="msg-reactions">${Object.entries(gr).map(([e,c])=>`<span class="reaction-chip">${e}<span>${c>1?c:''}</span></span>`).join('')}</div>`;
  }

  // Media message
  if (m.mediaUrl) {
    const mediaDiv=document.createElement('div'); mediaDiv.className='message-row '+(isSent?'sent':'received');
    const wrap=document.createElement('div'); wrap.className='msg-media';
    if (m.mediaType==='video') {
      wrap.innerHTML=`<video src="${m.mediaUrl}" controls playsinline></video>`;
    } else {
      wrap.innerHTML=`<img src="${m.mediaUrl}" alt="media" loading="lazy">`;
    }
    const meta=document.createElement('div'); meta.className='msg-meta';
    meta.innerHTML=`<span class="msg-time">${fmtMsgTime(m.ts)}</span>${isSent?seenIcon(m.seen):''}`;
    mediaDiv.appendChild(wrap); mediaDiv.appendChild(meta);
    return mediaDiv;
  }

  // Audio message

function startRecording(e) {
  if (isRecording) return;
  if (!navigator.mediaDevices) { toast('Microphone not available'); return; }
  navigator.mediaDevices.getUserMedia({audio:true}).then(stream => {
    isRecording=true; audioChunks=[];
    document.getElementById('chat-audio-btn').classList.add('recording');
    mediaRecorder=new MediaRecorder(stream);
    mediaRecorder.ondataavailable=e=>audioChunks.push(e.data);
    mediaRecorder.onstop=async()=>{
      stream.getTracks().forEach(t=>t.stop());
      const blob=new Blob(audioChunks,{type:'audio/webm'});
      const reader=new FileReader();
      reader.onloadend=async()=>{
        const dataUrl=reader.result;
        const msgRef=db.ref(`messages/${currentChatId}`).push();
        await msgRef.set({id:msgRef.key,audioUrl:dataUrl,senderId:currentUser.uid,senderName:currentProfile.displayName,ts:Date.now(),seen:false});
        db.ref(`unreadCounts/${currentChatPartner.uid}/${currentChatId}`).transaction(v=>(v||0)+1);
      };
      reader.readAsDataURL(blob);
    };
    mediaRecorder.start();
  }).catch(()=>toast('Cannot access microphone'));
}
function stopRecording() {
  if (!isRecording||!mediaRecorder) return;
  isRecording=false;
  document.getElementById('chat-audio-btn').classList.remove('recording');
  if (mediaRecorder.state!=='inactive') mediaRecorder.stop();
}

async function startCall(type) {
  if (!currentChatPartner) return;
  if (isBlocked(currentChatPartner.uid)) { toast('You have blocked this user.'); return; }
  currentCallType=type; isCallInitiator=true;
  try {
    const constraints = type==='video' ? {video:{facingMode:'user'}, audio:{echoCancellation:true,noiseSuppression:true}} : {audio:{echoCancellation:true,noiseSuppression:true}};
    localStream=await navigator.mediaDevices.getUserMedia(constraints);
  } catch(e) {
    // Show helpful error
    if (e.name==='NotAllowedError') { toast('Permission denied. Please allow access in your browser settings.'); }
    else if (e.name==='NotFoundError') { toast('No microphone/camera found on this device.'); }
    else { toast('Could not start call: '+e.message); }
    return;
  }
  peerConnection=createPeerConnection();
  localStream.getTracks().forEach(t=>peerConnection.addTrack(t,localStream));
  if (type==='video') document.getElementById('local-video').srcObject=localStream;
  showCallViewUI(type,'outgoing');
  const offer=await peerConnection.createOffer();
  await peerConnection.setLocalDescription(offer);
  const callRef=db.ref(`calls/${currentChatPartner.uid}`).push();
  currentCallId=callRef.key;
  await callRef.set({ callId:currentCallId, callerId:currentUser.uid, callerName:currentProfile.displayName, callerUsername:currentProfile.username, type, offer:{sdp:offer.sdp,type:offer.type}, ts:Date.now() });
  callRef.on('value',snap=>{
    const data=snap.val();
    if (data&&data.answer&&peerConnection&&!peerConnection.remoteDescription) {
      peerConnection.setRemoteDescription(new RTCSessionDescription(data.answer)).catch(console.error);
    }
  });
  db.ref(`iceCandidates/${currentCallId}/callee`).on('child_added',snap=>{
    if (peerConnection) peerConnection.addIceCandidate(new RTCIceCandidate(snap.val())).catch(()=>{});
  });
  window._callStartTime=Date.now();
}

function createPeerConnection() {
  const pc=new RTCPeerConnection({iceServers:ICE_SERVERS,iceCandidatePoolSize:10});
  pc.onicecandidate=e=>{
    if (e.candidate&&currentCallId) db.ref(`iceCandidates/${currentCallId}/${isCallInitiator?'caller':'callee'}`).push(e.candidate.toJSON());
  };
  pc.ontrack=e=>{
    if (!remoteStream) remoteStream=new MediaStream();
    remoteStream.addTrack(e.track);
    if (currentCallType==='video') document.getElementById('remote-video').srcObject=remoteStream;
    else document.getElementById('remote-audio').srcObject=remoteStream;
  };
  pc.onconnectionstatechange=()=>{
    const s=currentCallType==='video'?document.getElementById('vc-call-status'):document.getElementById('voice-call-status');
    const states={'connected':'Connected','connecting':'Connecting…','disconnected':'Reconnecting…','failed':'Connection failed'};
    if (s&&states[pc.connectionState]) s.textContent=states[pc.connectionState];
    if (pc.connectionState==='connected') window._callStartTime=window._callStartTime||Date.now();
  };
  return pc;
}

}

async function endCall() {
  const dur=window._callStartTime?Math.floor((Date.now()-window._callStartTime)/1000):0;
  stopRingtone();
  if (peerConnection) { peerConnection.close(); peerConnection=null; }
  if (localStream) { localStream.getTracks().forEach(t=>t.stop()); localStream=null; }
  remoteStream=null;
  document.getElementById('video-call-view').classList.remove('active');
  document.getElementById('voice-call-view').classList.remove('active');
  if (currentCallId) {
    if (isCallInitiator) db.ref(`calls/${currentChatPartner?.uid}/${currentCallId}`).remove();
    else db.ref(`calls/${currentUser.uid}/${currentCallId}`).remove();
    db.ref(`iceCandidates/${currentCallId}`).remove();
    if (currentChatPartner) {
      const entry={partnerUid:currentChatPartner.uid,partnerName:currentChatPartner.displayName,type:currentCallType,status:dur>0?'ended':'missed',duration:dur,ts:Date.now()};
      db.ref(`callLogs/${currentUser.uid}`).push(entry);
      db.ref(`callLogs/${currentChatPartner.uid}`).push({...entry,partnerUid:currentUser.uid,partnerName:currentProfile.displayName});
    }
    currentCallId=null;
  }
  window._callStartTime=null;
}

function playRingtone() {
  stopRingtone();
  ringtoneCtx=new (window.AudioContext||window.webkitAudioContext)();
  function beep() {
    if (!ringtoneCtx) return;
    const osc=ringtoneCtx.createOscillator();
    const gain=ringtoneCtx.createGain();
    osc.connect(gain); gain.connect(ringtoneCtx.destination);
    osc.frequency.value=880; osc.type='sine';
    gain.gain.setValueAtTime(0.4,ringtoneCtx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001,ringtoneCtx.currentTime+0.3);
    osc.start(ringtoneCtx.currentTime); osc.stop(ringtoneCtx.currentTime+0.3);
    if ('vibrate' in navigator) navigator.vibrate([200,100,200]);
  }
  beep(); ringtoneInterval=setInterval(beep,900);
}
function stopRingtone() {
  if (ringtoneInterval) { clearInterval(ringtoneInterval); ringtoneInterval=null; }
  if (ringtoneCtx) { ringtoneCtx.close().catch(()=>{}); ringtoneCtx=null; }
  if ('vibrate' in navigator) navigator.vibrate(0);
}

async function sendPushNotification(toUid, fromName, body) {
  // Write to DB; service worker reads this via push subscription
  await db.ref(`notifications/${toUid}`).push({ fromName, body, ts: Date.now() });
}

// ══════════════════════════════════════════════
function registerServiceWorker() {
  if ('serviceWorker' in navigator) {
    navigator.serviceWorker.register('./service-worker.js')
      .then(reg => { console.log('SW registered:', reg.scope); })
      .catch(err => console.log('SW error:', err));
  }
}
